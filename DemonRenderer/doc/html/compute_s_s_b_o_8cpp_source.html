341void ComputeSSBO::onUpdate(float timestep)
342{
343 if (m_window.isKeyPressed(GLFW_KEY_Q)) spdlog::info("You pressed Q");
344
345 // Update scripts
346 for (auto it = m_mainScene->m_actors.begin(); it != m_mainScene->m_actors.end(); ++it)
347 {
348 it->onUpdate(timestep);
349 }
350
351 // Update camera and its position in UBO
352 auto& camera = m_mainScene->m_actors.at(cameraIdx);
353
354 auto& pass = m_mainRenderer.getRenderPass(0);
355
356 pass.camera.updateView(camera.transform);
357 pass.UBOmanager.setCachedValue("b_camera", "u_view", pass.camera.view);
358 pass.UBOmanager.setCachedValue("b_camera", "u_viewPos", camera.translation);
359
360 // Update skybox camera
361 auto& skybox = m_mainScene->m_actors.at(skyboxIdx);
362 skybox.material->setValue("u_skyboxView",glm::mat4(glm::mat3(pass.camera.view)));
363}
