11 float DistanceOBBToPoint(Scene* scene, const OBBCollider& obb, const glm::vec3& point)
12 {
13 auto& obbActor = scene->m_actors.at(obb.actorIdx);
14
15 // ClosestPoint is the closest point inside or on the interior of the OBB
16 glm::vec3 closestPoint = obbActor.translation; // Set closest point to actor position
17 glm::vec3 centre = point - obbActor.translation; // Move the problem so actor is at world centre
18
19 // Get axis of OBB
20 glm::vec3 OBBRight = { obbActor.transform[0][0], obbActor.transform[0][1], obbActor.transform[0][2] };
21 glm::vec3 OBBUp = { obbActor.transform[1][0], obbActor.transform[1][1], obbActor.transform[1][2] };
22 glm::vec3 OBBForward = { -obbActor.transform[2][0], -obbActor.transform[2][1], -obbActor.transform[2][2] };
23
24 // Project along each axis, moving closest point as we go
25
26 // Project along X
27 float distX = glm::dot(closestPoint, OBBRight);
28 // Clamp to OBB half extents
29 distX = std::clamp(distX, -obb.halfExtents.x, obb.halfExtents.x);
30 // Move closest point
31 closestPoint += OBBRight * distX;
32
33 // Project along Y
34 float distY = glm::dot(closestPoint, OBBUp);
35 // Clamp to OBB half extents
36 distY = std::clamp(distY, -obb.halfExtents.y, obb.halfExtents.y);
37 // Move closest point
38 closestPoint += OBBUp * distY;
39
40 // Project along Z
41 float distZ = glm::dot(closestPoint, OBBForward);
42 // Clamp to OBB half extents
43 distZ = std::clamp(distZ, -obb.halfExtents.z, obb.halfExtents.z);
44 // Move closest point
45 closestPoint += OBBUp * distZ;
46
47 // Find the distance squared
48 float distanceSquared = glm::dot(closestPoint - point, closestPoint - point);
49
50 return sqrtf(distanceSquared);
51 }
