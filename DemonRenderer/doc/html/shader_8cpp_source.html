265bool Shader::compileShader(GLuint& shaderUnit, const std::filesystem::path& sourcePath, GLuint shaderType)
266{
267 bool result = false;
268
269 std::fstream handle(sourcePath.c_str(), std::ios::in);
270 if (!handle.is_open())
271 {
272 spdlog::error("Failed to open shader file: {0}", sourcePath.string());
273 return result;
274 }
275
276 std::stringstream sStream;
277 sStream << handle.rdbuf();
278 handle.close();
279
280 shaderUnit = glCreateShader(shaderType);
281 std::string tmp = sStream.str();
282 const GLchar* source = tmp.c_str();
283
284 glShaderSource(shaderUnit, 1, &source, 0);
285 glCompileShader(shaderUnit);
286
287 GLint isCompiled = 0;
288 glGetShaderiv(shaderUnit, GL_COMPILE_STATUS, &isCompiled);
289 if (isCompiled == GL_FALSE)
290 {
291 GLint maxLength = 0;
292 glGetShaderiv(shaderUnit, GL_INFO_LOG_LENGTH, &maxLength);
293

296 spdlog::error("Shader compile error: {0}", std::string(infoLog.begin(), infoLog.end()));
