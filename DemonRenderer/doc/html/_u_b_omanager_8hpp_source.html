31inline bool UBOManager::setCachedValue(const std::string& blockName, const std::string& uniformName, T value)
32{
33 auto it = m_UBOMap.find(blockName);
34 if (it != m_UBOMap.end())
35 {
36 auto& ubo = m_UBOs[it->second];
37 auto layout = ubo->getLayout();
38 for (auto& elem : layout)
39 {
40 if (elem.m_name == uniformName)
41 {
42 UniformData data;
43 data.type = elem.m_dataType;
44 data.enabled = true;
45 data.data = value;
46 std::pair<std::string, UniformData> dataPair(uniformName, data);
47 bool set = false;
48 for (auto& cacheData : m_UBODataCache) {
49 auto& cachedBlockName = cacheData.first;
50 auto& cachedUniformName = cacheData.second.first;
51 if (cachedBlockName == blockName && cachedUniformName == uniformName) {
52 cacheData.second.second = data;
53 set = true;
54 }
55 }
56 if (!set) m_UBODataCache.insert(std::pair<std::string, std::pair<std::string, UniformData>>(blockName, dataPair));
57 return true;
58 }
59 }
60 }
61
62 return false;
63}
